handleSubmit = (e) => {
    e.preventDefault();
    const { isAdd, form, record, maintainQ: { qualificationType } } = this.props;
    form.validateFields(async (errors, values: IQualification) => {
      if (errors) {
        return;
      }
      const params:any = {
        ...values,
        qualificationType,
        parentId: 0,
        isFolder: 0,
        files: record.files instanceof Array && record.files.length>0 ? record.files : []
      };

      if(!isAdd){
        params.id = record.id
      }

      const formData = new FormData();

      Object.keys(params).forEach((k: string)=>{
         if( params[`${k}`] instanceof Array){
           formData.append('filesStr', JSON.stringify(params[`${k}`]))
         }else{
           formData.set(k, params[`${k}`]);
         }
      })
      const { fileList } = this.state;
      fileList.forEach(file => {
        formData.append('upFiles', file);
      });

     await this.handleSave(formData)
    });
  };
